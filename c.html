package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main(){
    if err := ReadInputs(bufio.NewScanner(os.Stdin)); err != nil{
        fmt.Println(err)
    }
}

func ReadInputs(scanner *bufio.Scanner) error{

    scanner.Scan()
    numberOfTestCases,err := strconv.Atoi(scanner.Text())

    if err != nil || numberOfTestCases <= 0 {
        return fmt.Errorf("Number of test cases (N) must be integer and greater than 0")
    }

    return ReadTestCases(numberOfTestCases, scanner, []int{})
}

func ReadTestCases(remainingTestCases int, scanner *bufio.Scanner, result []int) error{

    if remainingTestCases <= 0 {
        PrintResults(result)
        return nil
    } else {

        scanner.Scan()
        amountOfNumber,err := strconv.Atoi(scanner.Text())
        if err != nil || amountOfNumber <= 0 {
            return fmt.Errorf("Amount of numbers for calculation (X) must be integer and greater than 0")
        }

        scanner.Scan()
        listOfNumber := strings.Split(scanner.Text()," ")

        if len(listOfNumber) < amountOfNumber {
            amountOfNumber = len(listOfNumber)
        }

        resval,err := SquareSum(listOfNumber[:amountOfNumber])
        if err != nil {
            return err
        }

        result = append(result,resval)

        remainingTestCases--;
        return ReadTestCases(remainingTestCases, scanner, result)
    }
}

func PrintResults(data []int) {

    if len(data) > 0 {
       fmt.Println(data[0])
       PrintResults(data[1:len(data)])
    }
}

func SquareSum(inputList []string) (int, error){

    if len(inputList) <= 0 {
        return 0, nil
    }

    num,err := strconv.Atoi(inputList[0])
    if err != nil {
        return 0, fmt.Errorf("List of number for calculation must consists of integer")
    }

    if num <= 0 {
        return SquareSum(inputList[1:])
    }

    resval,err := SquareSum(inputList[1:])
    if err != nil {
        return 0,err
    }

    return (num * num) + resval, nil
}
